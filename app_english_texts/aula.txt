4. Pacote html
O pacote html permite extrair texto de uma página HTML (parser).

import 'package:html/parser.dart' as parser; // Alias para evitar conflitos
import 'package:html/dom.dart';

void main() {
  // 1. Simulação do HTML bruto que viria da Wikipedia
  // Note que ele contém tags como <b> (negrito) misturadas ao texto.
  String wikipediaHtml = """
    <html>
      <body>
	<h1>Chole bhature</h1>
	<div id="content">
	   <p class="resumo">
	     <b>Chole bhature</b> is a food dish popular in the Northern areas of the Indian subcontinent.
	   </p>
	   <p>
	     It is a combination of chana masala and bhatura/puri.
	   </p>
	</div>
      </body>
    </html>
  """;

  // 2. O método parse() transforma a String gigante em um objeto navegável (DOM)
  Document document = parser.parse(wikipediaHtml);

  // 3. Buscando o primeiro parágrafo
  // O seletor 'p' busca a primeira tag <p> que encontrar no documento.
  Element? paragraphElement = document.querySelector('p');

  if (paragraphElement != null) {
    // 4. A propriedade .text remove automaticamente as tags internas (como <b>)
    // e retorna apenas o conteúdo textual.
    String cleanText = paragraphElement.text;

    print('--- Texto Extraído ---');
    print(cleanText); 

    // TODO: use .split(' ') para criar os botões
  } else {
    print('Nenhum parágrafo encontrado.');
  }
}

Podemos usar split também para analisar arquivos CSV:

void main() {
  // 1. Simulando o conteúdo de um arquivo CSV
  // Formato: Nome,Nota1,Nota2
  String csvData = """
Nome,Nota1,Nota2
Alice,8.5,9.0
Bruno,7.0,6.5
Carla,10,9.5
Daniel,5.5,6.0
""";

  // 2. Primeiro, separamos o texto em LINHAS
  // O trim() remove espaços vazios no início e fim do arquivo antes de dividir
  List<String> linhas = csvData.trim().split('\n');

  print('--- Boletim Escolar ---');

  // 3. Iteramos sobre as linhas.
  // Começamos de i = 1 para pular o cabeçalho ("Nome,Nota1,Nota2")
  for (int i = 1; i < linhas.length; i++) {
    String linhaAtual = linhas[i];

    // 4. Agora separamos as COLUNAS usando a vírgula
    List<String> colunas = linhaAtual.split(',');

    // O resultado de colunas será algo como: ['Alice', '8.5', '9.0']
    String nome = colunas[0];

    // 5. Convertendo as notas de String para Double para fazer conta
    double nota1 = double.parse(colunas[1]);
    double nota2 = double.parse(colunas[2]);

    // 6. Cálculo da média
    double media = (nota1 + nota2) / 2;

    // 7. Exibindo o resultado formatado (1 casa decimal)
    print('Aluno: $nome | Média: ${media.toStringAsFixed(1)}');
  }
}

8. Wrap
O Widget Wrap permite adicionar os botões e se não houver mais espaço ele automaticamente envia o item para a linha de baixo.

import 'package:flutter/material.dart';

void main() {
  runApp(MaterialApp(home: WikipediaPage()));
}

class WikipediaPage extends StatelessWidget {
  // 1. O texto que viria do seu parser HTML
  final String textoWikipedia = 
      "Chole bhature is a food dish popular in the Northern areas of the Indian subcontinent";

  @override
  Widget build(BuildContext context) {
    // 2. Transformando a frase em uma lista de palavras
    List<String> palavras = textoWikipedia.split(' ');

    return Scaffold(
      appBar: AppBar(title: Text("English with Wikipedia")),
      body: Padding(
	padding: const EdgeInsets.all(16.0),

	// 3. O WIDGET WRAP
	child: Wrap(
	  spacing: 8.0, // Espaço horizontal entre as palavras
	  runSpacing: 4.0, // Espaço vertical entre as linhas
	  children: palavras.map((palavra) {

	    // 4. Criando um botão para cada palavra
	    return ActionChip(
	      label: Text(palavra),
	      onPressed: () {
		// Aqui você chamaria a API do dicionário
		print("Buscando definição para: $palavra");
	      },
	      backgroundColor: Colors.blue[50],
	    );

	  }).toList(),
	),
      ),
    );
  }
}

import 'package:flutter/material.dart';

void main() => runApp(const MinimalTapApp());

class MinimalTapApp extends StatelessWidget {
  const MinimalTapApp({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      home: Scaffold(
	body: Center(
	  child: SimpleWordTapper(word: 'definition'),
	),
      ),
    );
  }
}

class SimpleWordTapper extends StatelessWidget {
  final String word;

  const SimpleWordTapper({required this.word, super.key});

  @override
  Widget build(BuildContext context) {
    // 1. O GestureDetector envolve o widget que você quer tornar interativo.
    return GestureDetector(
      // 2. onTap é o callback chamado quando o usuário toca.
      onTap: () {
	// Ação: Abrir a página de definição
	print('A palavra "$word" foi clicada.');
      },
      // 3. O widget visual (child) é o texto da palavra.
      child: Text(
	word,
	// Adicionando um estilo visual sutil (sublinhado) para que o usuário saiba que é clicável.
	style: const TextStyle(
	  color: Colors.blue,
	  decoration: TextDecoration.underline,
	),
      ),
    );
  }
}
